#  Types conversion

//  скоставвали число в строку / ПРЕОБРАЗОВАНИЕ ЧИСЛА В СТРКОУ
```js
let value = 10; // number

value = value + ""; // string (1й способ?)
value = `${value}`; // с помощью такой записи (обратыные кавычки, зщак доллара и скобки)  (2й споосб?) // эта зхапись удобна тем что мы мождем дальше продолжить едлать строку прямо в кавычках value = `${value} привет`;
value = value.toString(); // (3й споосб?), с помощью метода toString()
value = String(value); // (4й споосб?), с помощью вызова глобального объекта String и помезщением туда числа
```
- в джс есть прототипное наследование (по факту в джс все объекты)
  
//  ПРЕОБРАЗОВАНИЕ СТРОКИ В ЧИСЛО
```js
let value = "10"; // string

value = Number(value); // 1st (с помощью метода Number() с переданной строкой)

value = +value; // 2st (с помощью унарного оператора + перед строкой)

value = parseInt(value); // 3st с помошью parseInt() 

// различие parseInt() от Number()
let value = "10a";
value = Number(value); // NaN
value = parseInt(value); // number!! (parseInt более сербезная функция в отличии от Number)

let value = "10.333"; // 4st с помошью 
value = parseFloat(value); // тоже самое что и parseInt() но для дробных чисел / чисел с плавающей запятой (если бы здесь использовали parseInt то резальтат был бы не 10.333 а просто 10 (без дробной части))
```

- дополнительно
  ```js
  let number = parseFloat(10.12345.toFixed(3)); // toFixed() - сколько чисел мы хотим оставить после запятой (в дробнйо части) / 10.12345.toFixed(3); = 10.123 (3 числа оставили)
  ```


# Types casting (неявное приведение?? - в первой части оно уже было показано)

```js
value = 10 + '10'; // 1010 string
value = 10 * '10'; // 100 number
value = 10 / '10'; // 1 number
value = 10 - '10'; // 0 number
value = 10 % '10'; // 0 number

value = 10 + true; // 11 number // т.к. true - 1, false - 0 ==> 10+1=11 (неявное приведение типов сработало)

value = 10 + null; // 10 (т.к. null это ничего)

value = 10 + undefined; // NaN (тк undefined не может привестись к числу)
 
value = 10 + NaN; // NaN 

value = 10 + []; // 10 number 
value = 10 + {}; // 10[object Object]

value = true + true; // 2 number


etc  
```
