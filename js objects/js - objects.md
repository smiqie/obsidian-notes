- object и псользуектся для хранения коллекций различных знаечний
- у объектов есть св-ства. по сути свва это пара `ключ: значение`, где ключ - строка (имя свойства) а значение можем быть любым типом данных
- работа с обектом
```js

/* ===== создание обхъекта ===== */
// 1 способ
const person = new Object();
// 2 способ


// (снизу 2 запипи + у обеъкта) - потсоянно используется в коде особенно когда мы отпарвляем какието даннеы на сервер и формируем объект (мы получаем какието аргументы в функцию или у нас есть какието константы/перемкнные и мы создаем объекти чтобы не писать id: "sdfdsf" мы просто созадем перемнную id со значением и в объекте пишем короткую запись id  ) 
const hobby = "bobbi lmao"; // 1й варик динамики - релаьный рпимер из разарбьотки (создние динамических клюбчей)
const id = "13#smI!sd@"// (нпр какотйо сгенерируемым адишник) 2й варик динамики (наиболее популярный) - релаьный рпимер из разарбьотки(создние динамических клюбчей)

const person = {
	name: "Dima",
	age: 33, 
	isDeveloper: true,
	// свойство но вложенный объект
	address: {
		country: "Serbia",
		city: "Sombor",
		street: "Branka Radicivica 19"
	},
	// та кобычно не делюат !!! пдохая практика
	// под каоптом 0 преобразуется в строку и обьратиться к нему мы смодем ток 2мя способами
	//1. person[0] 2. person["0"]
	0: "just 0",
	// если ключ занимает пару слов то его нужно писать в двонйыхх кавычках, и оьращаться к нему можно ток ... способами - нпр person["last name"] - это называется составное свойство
	"last name": "Kora",
	// ллжим в наш объект функцию
	greet: function(){
		console.log("Hello, Im Dima");
	},
	// создание стрелочной функции в объекте
	greetSecond () => console.log("Hello, Im Dmitry"), 
	// создание ключа с названием hobby (ps вообще не понял это катко свЗано с перменной сверху - атвор ее создал перед тем как ключ добавить) и че за квадратные скобки зачем тип - он ячет сказал про динамичсекое создание
	[hobby]: "runnibg, skiing, coding",
	// без квадратных скобок прописываем айди (когда клюбч и значение свопадают мы можем просто анписать 1 слово - джс автоматом рассмотрит это так - что у нас создаться ключ id и значерие возмется из его перемнной, т.е. по сути id == id: "13#smI!sd@" == id: id)
	id
}

/* ===== работа с обхъекта ===== */

// получаем сам объект
console.log(person); 


// получаем отдельное св-во объекта 
console.log(person.age); 

// олучаем отдельное св-во объекта через ["имя свойства"]
console.log(person["name"]); 

// получаем отдельное св-во объекта - вложенный объект (по сути получаем получаем сам объект address, который вложен как своййство в объект person) - ъх как кратко и понятно сформулирвоать
console.log(person.address); 
// образение к отдельному свойству address объекта person и послждеующее обращение к свойству объекта address - country
console.log(person.address.country); 


// обращение к несуществующему свойству isPositive
console.log(person.isPositive); // undefined

// обращение к несуществующему свойству isPositive и полсеющему образению к еще одну несовущствующему свойству у свойства isPositive - ъх как номр сформулирвоать
console.log(person.isPositive.qwerty); // TypeError
	// изюегание ошибки TypeError с помощью ? (не оч хорошая практика  - но испоьзуются часто)
	console.log(person.isPositive?.qwerty); // undefined
	

---

// удаление какогото свойства у объекта с помощбю ключевого слова delte

delete person[0]; // прописываем имя свойства
person[0]; // undefined

---

// вызов функции у объекта
person.greet();
person.greetSecond();

различие между использование функции в объекте с клюбчевым словом function либо стрелочную функцию, когда мы функцию присваиваем как знаечние ключа (чутка позже обсудим)


// копируем ЗНАЧЕНИЕ из объекта в перменную prevName
const prevName = person.name;

// изменяем знеачние СВОЙСТВА объекта (не сомтря на то-  ч то сам объект const), А НЕ САМ ОБЪЕКТ - те если бы мы хотели изменить сам объект то была бы оишбка
person.name = "Alex";

// динамическое создание ключа/свойства у объекта и присваивание ем знаечния
person.isHappy // undefined
person.isHappy = true; // динамическое создание пары ключ-значение у объекта
person.isHappy // true



// проаверка наличия свойства в объекте (чтобы получить тру или фолс)
console.log("age" in person) // есть ли свойство age в объекте person? true


// создание новго объекта с использованием страого (рассмотрим позже)
const newPerson = person; // присваиваем ссылка на объект person
console.log(newPerson === person); // true
```
