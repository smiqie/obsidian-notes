```css
@keyframes move-to-right{
	/* пишем группы правил, кадое из которых отвечеат за определнный момент врмеени выполнения анимации*/
	from{
		translate: 0;
	}
	to{
		translate: 75vw;
	}
}

@keyframes move-to-right{
	/* название такой группы правил мождет быть либо клбчевое слово from/to - что соотсвеутвует начальному и конечному кадру анимации ИЛИ задать значение в процентах от 0% до 100%  */
	0%{
		translate: 0;
	}
	100%{
		translate: 75vw;
	}
}

/* начальный ключевой кадр задать не обязательно, елси его не здаать то за первый клбчевой кадр будет взято изначальное сотсояние св-в элемента. ключевых кадров модем добавлять скок угождно*/
@keyframes move-to-right{
	100%{
		translate: 75vw;
	}
}
```
- @keyframes - объявдение груп правил ключевых кадров, которые описываются в директиве @keyframes
- ключевые кадры описывают значения св-в в опредленный промежуток времени выполнения анимации

1. animation-name: move-toi-right; // имя анимации
2. animation-duration: 1s; длительнотсь анимации в секунлах (s) или милоискенудлазх (ms). значение по дкофлту - 0 сек
3. animation-delay: 1s; задержка пережд началом анимки (секнудлы или милоисекунды) / мождно указывтаь отрицательные знаечния чтобы анимация запускалась как бдуто в фоне и фактически анчнется не с самого начала (+ мб нужна чтобы наш сайт успел прогрузиться)
4. animation-iteration-co8unt: 3; по умолчанию анимации раюботают 1 раз, но это значения можно изменить / в его значение можно укзаать целое число больше 0 / иили ключевое слово infinite - анимка будет бесокнечная
5. animation-timing-function: ease; згначения  аналогично transition, фукнкция сокркости анимации
6. animation-direction: normal; говорим браузеру о том должна ли анимка выполняться в обратном порядке  (def - normal --> анимка будет выполняться от 1го до послденго клюбчевого кадра а затем состоние элемента вернется к 1му кадру) // значение reverse --> то ключевые кадры анимки будут выполняться в другом порядке с последнего до первого затем анимка вернется к послднему кадру // alternate - нечетный повтор анимаций 1й 3й 5й и тд будут выполняться в обыкновенном порядке а кадлый четный повтор 2й 4й и тд будут воспроизводится в обратном порядке  // alternate-reverse - аналогично пршлому, но в обратном пордке будут произвождитсяь уже нечтные кадры 1й 3й и тд
animation-play-state: - позволяет ставить выполнение анимации ан паузу. значение по умолчанию running -->  // paused-  при котормо анимация зщамораждивается т.е. ставится на паузу (с вомзщожностью дальнешего продолжения ее - зависит от обстоятельст нпр с ховер эффектом)
